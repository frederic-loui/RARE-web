```
hostname r1
buggy
!
logging file debug ../binTmp/zzz-log-r1.run
!

!

 exit
!
router pvrp4 1
 vrf v1
 router-id 4.4.4.1
 redistribute connected
 exit
!
router pvrp6 1

 router-id 6.6.6.1
 redistribute connected
 exit
!

 no description

 no log-link-change

!

!
```
